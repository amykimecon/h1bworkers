\section{Introduction}
\begin{frame}
    \frametitle{Motivation}
    \begin{itemize}
        \item Immigrants increasingly crucial to skilled workforce
        \begin{itemize}
            \item 59\% of PhD computer scientists in U.S. born abroad \textcolor{gray}{[Zwetsloot et al. 2019]}
        \end{itemize}
        \vspace{4mm}
        \item \textbf{H-1B visas}: tied to employer-worker pair, allocated via lottery
        \begin{itemize}
            \item H-1B lottery applications in 2024 exceeded availability by 5.6x 
        \end{itemize}
        \vspace{4mm}
        \item Policy tradeoff: filling high-skill jobs without crowding out native workers
        \begin{itemize}
            \item Missing element: what happens to \textbf{worker that \textit{loses} the H-1B visa}?
            \item Important for global efficiency + policy counterfactual 
        \end{itemize}
        \vspace{4mm}
        \item \textcolor{maincolor}{\textbf{This Project:}} Novel data + linkage strategy $\rightarrow$ first to estimate worker-level effects of losing H-1B lottery
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Project Overview}
    Empirical hurdles in estimating worker-level H-1B effects:
    \vspace{2mm}
    \begin{enumerate}
        \item Lack of information on H-1B lottery losers or activity outside U.S. 
        \begin{itemize}
            \item \textbf{This Project:} New data on universe of H-1B applicants + LinkedIn profiles
        \end{itemize}
        \vspace{2mm}
        \item No personal identifiers (name, ID number) in H-1B data 
        \begin{itemize}
            \item \textbf{This Project:} Probabilistic linking across many sparse fields (including firm) 
        \end{itemize}
        \vspace{2mm}
        \item Don't observe workers' outcomes unless they win visa 
        \begin{itemize}
            \item Traditional view of H-1B: firm hires worker if they win visa, otherwise don't 
            \vspace{2mm}
            \item \textbf{Today:} In reality, majority of H-1B applicants are \textit{already working at firm} on a different temporary visa
            \begin{itemize}
                \item New view of H-1B: firm employs worker who stays if they win visa, otherwise leaves 
            \end{itemize}
            \vspace{2mm}
            \item \textbf{This Project:} Link by subsetting to workers already at firm at time of application
        \end{itemize}
    \end{enumerate}
    \vspace{2mm}
    \textbf{End Goal:} Individual-level panel data on 118k H-1B lottery applicants over four years
\end{frame}

\begin{frame}
    \frametitle{Research Questions}
    When workers lose the H-1B visa...
    \vspace{2mm}
    \begin{itemize}
        \item \textbf{Characterizing Losers:} where do they go?
        \begin{itemize}
            \item Return migration or other high-immigration destination?
            \item Switch occupations? Firms? Heterogeneity in response by worker?
        \end{itemize}
        \vspace{3mm}
        \item \textbf{Productivity/Entrepreneurship:} how does it affect their productivity?
        \begin{itemize}
            \item Limited employer mobility under H-1B\textemdash more entrepreneurship outside U.S.?
            \item In the works: linking to GitHub data for direct measure of productivity 
        \end{itemize}
        \vspace{3mm}
        \item \textbf{Worker Outcomes:} how does it affect their wages?
        \begin{itemize}
            \item Lottery loss is exogenous shock to employment
            \item Decomposition into effect of leaving firm + effect of leaving U.S.
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Background and Context}
\begin{frame}\label{h1b_visa_allocation}
    \frametitle{What is an H-1B Visa?}
    \begin{itemize}
        % \item Stated Purpose: ``to help employers who \textbf{cannot otherwise obtain needed business skills and abilities from the U.S. workforce}...''
        % \vspace{3mm}
        \item Temporary work visa (six years) for workers in `specialty occupations'
        \begin{itemize}
            \item ``...requires the application of a body of \textbf{highly specialized knowledge} and the attainment of at least a bachelor’s degree...''
            \item Employment-based visa: employer must apply on worker's behalf 
            \item Does not lead directly to permanent residency but employer can sponsor green card
            \item Portable between employers but (1) no self-sponsorship and (2) cannot transfer green card application
        \end{itemize}
        \vspace{2mm}
        \item Capped at 85,000 visas annually since 2004
            \begin{itemize}
                \item 20k for `advanced degree exempt' (ADE) with M.A.+ from U.S. school
                \item Demand has exceeded cap every year since 2004 \hyperlink{h1b_apps_by_year}{\beamergotobutton{Applications by Year}} 
                % \item 20k for `advanced degree exempt' (ADE) with M.A.+ from U.S. school
                % \item Non-profits and schools exempt from cap
            \end{itemize}
        \vspace{2mm}
        \item Since 2013 (and in 2007 and 2008) USCIS allocates visas using a \textbf{lottery} \hyperlink{timeline}{\beamergotobutton{Timeline}} 
            \begin{itemize}
                \item Other years: cut off applications on day that cap is reached
            \end{itemize}
    \end{itemize}
\end{frame}
% ``A specialty occupation is one that requires the application of a body of \textbf{highly specialized knowledge} and the attainment of at least a bachelor’s degree or its equivalent. The intent of the H-1B provisions is to help employers who \textbf{cannot otherwise obtain needed business skills and abilities from the U.S. workforce...}''

\begin{frame}
    \frametitle{How can employers hire workers without an H-1B visa?}
    \textbf{Primary Method:} F-1 Optional Practical Training (OPT)
    \begin{itemize}
        \item F-1: student visa (tied to enrollment)
        \item OPT: twelve months of authorized employment after graduation 
        \item OPT STEM Extension: STEM graduates can extend OPT to total of 36 months 
        \item \textbf{Common Strategy:} Begin work immediately after graduation under OPT, apply for H-1B every year until OPT runs out (three chances)
    \end{itemize}
    \vspace{4mm}
    Other Methods 
    \begin{itemize}
        \item L visa (intercompany transfer): requires 2+ years at company location abroad
        \item TN visa: Canadian and Mexican `professionals'
        \item Direct to Green Card via OPT: Risky due to variable processing times
    \end{itemize}
\end{frame}

        \vspace{1mm}
        \item \textbf{Takeaway:} Large share already in U.S. on F-1 + in STEM (3y OPT-eligible) 
    \end{itemize}    
\end{frame}

\begin{frame}\label{linking}
    \frametitle{Linking Strategy: Overview}
    \textbf{Idea:} link subset of applicants already working at firm before H-1B application 
    \vspace{3mm}
    \begin{enumerate}
        \item Link Firms from H-1B to LinkedIn
        \begin{itemize}
            \item $\sim$60\% of firms currently linked on name, location, industry 
            \item Expect to link 80\% with more advanced linking
        \end{itemize}
        \vspace{3mm}
        \item Use firm linkages to determine potential matches in LinkedIn data
        \begin{itemize}
            \item Pool of potential matches for applicant $i$ from firm $j$: everyone working at $j$ \textit{prior to} H-1B lottery application (as reported on LinkedIn)
        \end{itemize}
        \vspace{3mm}
        \item Use probabilistic matching techniques to identify likely matches 
        \begin{itemize}
            \item Matching will be based on sex, year of birth, and country of birth
        \end{itemize}
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Linking Strategy: Imputing Information from LinkedIn}
    How do we determine sex, year of birth, country of birth from LinkedIn? 
    \vspace{2mm}
    \begin{itemize}
        \item \textbf{Sex:} impute gender from name
        \begin{itemize}
            \item Preloaded imputation appears to do poorly for East and South Asian names (can try different models based on country of birth)
        \end{itemize}
        \vspace{2mm}
        \item \textbf{Country of Birth:} two-pronged approach 
        \begin{itemize}
            \item Name: use `name2nat' package (ML model built by linguist, trained on Wikipedia)
            \item Education History: use location of earliest institution (15-20\% report high school)
        \end{itemize}
        \vspace{2mm}
        \item \textbf{Year of Birth:} impute from year ending high school or starting Bachelor's
    \end{itemize}
\end{frame}

\begin{frame}\label{firm_sample}
    \frametitle{Sample Construction: H-1B Applicants}
    Due to difficulty linking, main sample of H-1B applicants will drop:
    \vspace{2mm}
    \begin{enumerate}
        \item \textbf{Applicants from Large H-1B Employers:} Drop applicants from firms who submit 50+ lottery applications (66\%)
        \begin{itemize}
            \item Distribution of lottery applications: long right tail \hyperlink{app_hist_binned}{\beamergotobutton{Histogram}} 
            \item Top 20 employers submit 20\% of applications 
            \item Hard to match individuals
        \end{itemize}
        \vspace{2mm}
        \item \textbf{Applicants from Supsected Outsourcing/Staffing Companies:} Drop applicants from firms with over 50\% of applications marked as duplicates (41\%)
        \begin{itemize}
            \item Outsourcing companies: treat workers as interchangeable \& `overapply' for H-1Bs
            \item Staffing companies: use shell corps to submit multiple applications per worker
            \item Inaccurate picture of skilled immigrant workers at U.S. firms; harder to match
        \end{itemize}
    \end{enumerate}
    \vspace{2mm}
    \textbf{Main Sample:} 430k lottery applicants (over 138k employer x years)
\end{frame}

\section{Empirical Evidence on Pre-H-1B Visa Holders}
\begin{frame}
    \frametitle{How many workers will we be able to link?}
    Initial sample: 430k lottery applicants
    \vspace{3mm}
    \begin{enumerate}
        \item Subset to those already working for H-1B company before lottery (pre-H-1B visa workers)
        \begin{itemize}
            \item \textbf{Question:} What share of our sample of lottery applicants are already working for their H-1B company prior to the H-1B lottery?
        \end{itemize}
        \vspace{3mm}
        \item Match rate via probabilistic matching
        \begin{itemize}
            \item \textbf{Question:} What share of pre-H-1B visa workers can we match to LinkedIn?
        \end{itemize} 
    \end{enumerate}    
\end{frame}

\begin{frame}
    \frametitle{Upper bound on pre-H-1B visa holders: 84\%}
    Initial sample: 430k lottery applicants
    \vspace{3mm}
    \begin{enumerate}
        \item Subset to those already in U.S. on other visa (pre-H-1B visa holders)
        \begin{itemize}
            \item Among main sample, \textcolor{maincolor}{\textbf{84\%}} of winners with non-missing visa info (66\% of all winners) are on work-eligible visa
            \item \textbf{Upper bound} on true number (since U.S. students overrep among winners)
        \end{itemize}
        \vspace{3mm}
        \item Among pre-H-1B visa holders, subset to those working for H-1B company before lottery (pre-H-1B visa workers)
        \begin{itemize}
            \item \textbf{Question:} What share of H-1B applicants on different visa work for company before lottery result?
        \end{itemize}
        \vspace{3mm}
        \item Match rate via probabilistic matching
        \begin{itemize}
            \item \textbf{Question:} What share of pre-H-1B visa workers can we match to LinkedIn?
        \end{itemize} 
    \end{enumerate} 
\end{frame}

\begin{frame}

\begin{frame}
    \frametitle{Ideal Empirical Strategy}
    For lottery applicant $i$ at firm $j$ in lottery year $s$, model outcome $y$ in year $t$ as:
    $$y_{ijst} = \beta_0 + \beta_1^{reg} Win_{ijs} \cdot (1-ADE_i) + \beta_1^{ade} Win_{ijs}\cdot ADE_i + \varepsilon_{ijst}$$
    where $Win_{ijs}$ is an indicator for winning the lottery in year $s$ and $ADE_i$ is an indicator for ADE status
    \begin{itemize}
        \item Impute $ADE_i$ from LinkedIn data 
        \item Identification assumption satisfied by lottery structure: $E[\varepsilon_{ijst}|Win, ADE] = 0$
        \begin{itemize}
            \item As long as selection into linkage not dependent on win status or ADE status, internal validity satisfied 
            \item External validity more of an issue, will depend on what linkage looks like 
        \end{itemize}
        \item First stage: effect of lottery win on H-1B visa takeup (need to submit petition)
        \item Other outcomes: probability of staying at firm after $n$ years, probability of leaving U.S., imputed wage, etc. 
    \end{itemize}
\end{frame}


\begin{frame}\label{h1bfiling}
    \frametitle{H-1B Filing Process: Details}
        \begin{itemize}
            \item To hire H-1B worker, firm first required to file Labor Condition Application (LCA) with DOL attesting that the job will not replace/harm incumbent workers
            \begin{itemize}
                \item Not tied to a specific employee, must be filed no more than six months before start date
                \item Includes information about job type (occupation code), start and end dates, work location, and salary
            \end{itemize}
            \item Prior to FY2021, firms required to fill out full paperwork for H-1B (including LCA and filing fees of $\sim$\$500 per applicant). Only forms of lottery winners processed, rest thrown out. 
            \item Starting FY2021, firms only required to fill out short form and \$10 filing fee to enter lottery, then only lottery winners required to fill out full paperwork. 
            \begin{itemize}
                \item Bloomberg investigation\textemdash streamlined process caused fraud to skyrocket (multiple submissions of same employee from different companies)
            \end{itemize}
        \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Sources of Variation: Literature and Context}
    \begin{itemize}
        \item Quota of 65,000 established in 1990, increased in 1999 then again in 2001, reset to 65,000 + 20,000 ADE in 2004
        \begin{itemize}
            \item Older literature using these aggregate increases as source of variation \textcolor{gray}{\footnotesize [Kerr and Lincoln 2010, Peri, Shih and Sparber 2015, Bound, Khanna and Morales 2017]}
        \end{itemize}
        \item Initially, visas allocated on a first-come, first-served basis until cap reached 
        \item In FY2006-7, USCIS ran unannounced pilot lotteries for applicants on day that cap was reached
        \begin{itemize}
            \item Small sample of ~3k firms \textbf{but} includes data on lottery losers \textcolor{gray}{\footnotesize [Doran, Gelber and Isen 2022 (DGI)]}
        \end{itemize}
        \item In FY2008 (and most future years) the cap was reached on the first day of the filing period, so \textit{all} visas were distributed via lottery
        \begin{itemize}
                \item All firms \textbf{but} no data on lottery losers \textcolor{gray}{\footnotesize [Dimmock, Huang and Weisbenner 2022 (DHW); Mahajan et al. 2024 (MMSCB)]}
                \item Instead, DHW and MMSCB use LCAs as imperfect proxy for H-1B applications
        \end{itemize}
    \end{itemize}
\end{frame}
