\begin{frame}
    \frametitle{Merged Sample: Reducing `Multiplicity'}
    \begin{itemize}
        \item Baseline strategy: restrict H-1B applications to those from small, non-outsourcing firms and LinkedIn users to those more likely to be on OPT, then merge and filter 
        \vspace{2mm}
        \item \textbf{Problem:} `multiplicity' (number of potential matches per H-1B app) still too high  \vspace{2mm}
        \item \textbf{Last time:} use pre-determined rules to subset to H-1B applications we think will be easier to match (exclude China, India, Anglophone countries)
        \begin{itemize}
            \item Low external validity, also low internal validity?
        \end{itemize} \vspace{2mm}
        \item \textbf{Different approach:} Cherry-pick firms with low multiplicity \textit{after} merging (try to pick firms that have `ideal' experimental setup)
        \begin{itemize}
            \item Low external validity, high internal validity?
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Merged Samples}
    \begin{enumerate}
        \item \textbf{Baseline:} merge and filter with existing samples (78.9k H-1B Apps matched to 828k potential matches, multiplicity 10) \vspace{2mm}
        \item \textbf{`Pre-Filtered':} further filter the H-1B applications (before merging) to exclude applications from China, India, Anglo countries (15.1k H-1B Apps matched to 65.2k potential matches, multiplicity 4) \vspace{2mm}
        \item \textbf{`Post-Filtered':} filter baseline to only include companies with
        \begin{itemize}
            \item All applications matched with low multiplicity ($<$4)
            \item At least one win and one loss observed in lottery
            \item Sufficient diversity in LinkedIn user matches (don't want case where the same user is matched to every application)
        \end{itemize}
        (10.2k H-1B Apps matched to 23.3k potential matches, multiplicity 2)
    \end{enumerate}    
\end{frame}

\begin{frame}
    \frametitle{Empirical Strategy}
For application $i$ with company $c$ in lottery year $t$:
\begin{equation*}
    y_{ict} = \beta_1 win_{ict} + \beta_2 ADE_{ict} + \delta_{ct}  + \varepsilon{ict}
\end{equation*}
where $win_{ict}$ indicates winning the H-1B lottery, $ADE_{ict}$ is the estimated probability of an individual being eligible for the ADE lottery (higher chance of winning), and $\delta_{ct}$ are company x lottery year FEs.

\vspace{3mm}
Outcomes $y_{ict}$:
\begin{itemize}
    \item $work1yr$\textemdash Indicator for still working at company one year after lottery 
    \item $work2yr$\textemdash Indicator for still working at company two years after lottery (3+ years becomes problem for 2023 lottery)
    \item $update$ \textemdash Indicator for updated LinkedIn profile since lottery 
    \item $update2yr$ \textemdash Indicator for updated LinkedIn profile since 2yr after lottery 
\end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Potential Problems with Identification}
    \begin{itemize}
        \item Baseline and Pre-Filtered: within companies, not requiring that all applications are matched. Could be that it's easier to match winners and/or those who stay at company longer 
        \item Pre-Filtered: filtering out some countries. Could be that match rates differ by country (highly likely), and that probability of staying at company also differs by country 
        \item All: even within post-filtered, possible that winners are more likely to get `better matches' and better matches are also likely to have longer tenure at firm. Would bias results upward.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Takeaways}
    \begin{itemize}
        \item First results table suggests very small but significant increase in probability of still working at company one year after the lottery!
        \item But effect seems to disappear by two years post -- maybe because lottery winners also start to leave and are more likely to update their profile?
        \item \textbf{Fundamental Problem:} Zeros (or small estimates) could be true zeros or mismeasurement
    \end{itemize}
