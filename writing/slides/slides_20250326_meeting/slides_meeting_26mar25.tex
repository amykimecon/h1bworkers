\section{Overview}
\begin{frame}
    \frametitle{Overview}
    \textbf{Research Question:} do workers at H-1B winning firms have better outcomes than workers at H-1B losing firms?
    \begin{itemize}
        \item Key contributions: use new dataset (Bloomberg H-1B data merged with LinkedIn) to (1) answer crowd-out question better and (2) dig into new outcomes
        \item Empirical strategy: compare hires, promotions, exits at firms who win H-1B lottery vs. lose H-1B lottery
        \item Progress so far:
        \begin{enumerate}
            \item Descriptives and summary stats for both datasets 
            \item Merging firms\textemdash 65\% merged currently, can almost certainly do better but tabling for now
            \item Extremely preliminary results on crowd-out 
        \end{enumerate}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{H-1B Visas: Policy Overview}
    \begin{itemize}
        \item Nonimmigrant visa for workers in `specialty occupations' 
        \begin{itemize}
            \item ``A specialty occupation is one that requires the application of a body of \textbf{highly specialized knowledge} and the attainment of at least a bachelor’s degree or its equivalent. The intent of the H-1B provisions is to help employers who \textbf{cannot otherwise obtain needed business skills and abilities from the U.S. workforce...}''
        \end{itemize}
        \item Employment-based visa: employer must apply on worker's behalf
        \begin{itemize}
            \item 'H-1B portability' between employers introduced in 2000, but lots of friction remains
            \item 'Dual-intent' visa: employer can sponsor green card for worker 
        \end{itemize}
        \item Generally a three-year visa with a three-year extension
        \item Capped at 65,000 normal visas annually and 20,000 advanced-degree exempt (ADE) visas for U.S. advanced degree holders (some employers exempt from cap)
    \end{itemize}
\end{frame}%Literature, Context and Timeline}

\begin{frame}\label{h1bfiling}
    \frametitle{H-1B Filing Process}
        \begin{itemize}
            \item To hire H-1B worker, firm first required to file Labor Condition Application (LCA) with DOL attesting that the job will not replace/harm incumbent workers
            \begin{itemize}
                \item Not tied to a specific employee, must be filed no more than six months before start date
                \item Includes information about job type (occupation code), start and end dates, work location, and salary
            \end{itemize}
            \item Prior to FY2021, firms required to fill out full paperwork for H-1B (including LCA and filing fees of $\sim$\$500 per applicant). Only forms of lottery winners processed, rest thrown out. 
            \item Starting FY2021, firms only required to fill out short form and \$10 filing fee to enter lottery, then only lottery winners required to fill out full paperwork. 
            \begin{itemize}
                \item Bloomberg investigation\textemdash streamlined process caused fraud to skyrocket (multiple submissions of same employee from different companies)
            \end{itemize}
        \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Sources of Variation: Literature and Context}
    \begin{itemize}
        \item Quota of 65,000 established in 1990, increased in 1999 then again in 2001, reset to 65,000 + 20,000 ADE in 2004
        \begin{itemize}
            \item Older literature using these aggregate increases as source of variation \textcolor{gray}{\footnotesize [Kerr and Lincoln 2010, Peri, Shih and Sparber 2015, Bound, Khanna and Morales 2017]}
        \end{itemize}
        \item Initially, visas allocated on a first-come, first-served basis until cap reached 
        \item In FY2006-7, USCIS ran unannounced pilot lotteries for applicants on day that cap was reached
        \begin{itemize}
            \item Small sample of ~3k firms \textbf{but} includes data on lottery losers \textcolor{gray}{\footnotesize [Doran, Gelber and Isen 2022 (DGI)]}
        \end{itemize}
        \item In FY2008 (and most future years) the cap was reached on the first day of the filing period, so \textit{all} visas were distributed via lottery
        \begin{itemize}
                \item All firms \textbf{but} no data on lottery losers \textcolor{gray}{\footnotesize [Dimmock, Huang and Weisbenner 2022 (DHW); Mahajan et al. 2024 (MMSCB)]}
                \item Instead, DHW and MMSCB use LCAs as imperfect proxy for H-1B applications
        \end{itemize}
    \end{itemize}
\end{frame}

\section{H-1B Data: Descriptives and Challenges}
\begin{frame}
    \frametitle{Bloomberg Data: Summary}
    \begin{itemize}
        \item Covers four years of lotteries: FY2021 (March 2020 Lottery)-FY2024 (March 2023 Lottery) 
        \item Includes lottery application data and lottery status for all H-1B applicants \textit{subject to the lottery}
        \begin{itemize}
            \item Includes both regular and Advanced Degree Exempt (ADE) applicants (subject to different lotteries) \textbf{but does not identify the group into which the applicant falls}
            \item Does not include any information on cap-exempt employers (non-profits, higher ed, etc.), applications for change in employer or visa extension
            \item \textcolor{maincolor}{Contrast to previous FOIA data} which has no information on lottery losers 
        \end{itemize}
        \item Additionally, includes information from I-129 (H-1B Visa Application Form) for lottery applicants who won \textbf{and} submitted form (66\% of winners)
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Company Sample Selection}
    \begin{itemize}
        \item Very long right tail
        \begin{itemize}
            \item Top 20 employers submit 20\% of applications
        \end{itemize}
        \item Well-documented issue with outsourcing/IT staffing companies flooding H-1B lottery 
        \item Want a sample of small companies most likely to see effect of H-1B win on workers and not dependent on H-1Bs
        \item \textbf{Preferred Sample:} Non-staffing companies with fewer than 50 H-1B lottery applications [138k emp x lottery years (80\% of total employers; 24\% of total applications)]
        \begin{itemize}
            \item Staffing companies IDed as those with at least 50\% of applications marked as duplicate entries (can probably do better)
            \item $<4$\% of employers submit 50 or more applications 
            \item Preferred sample [PS] has much higher win rate (36\% vs. 30\% for out of sample [OOS]), likely from higher share of ADE applicants
            \item Applicants from OOS firms overwhelmingly from India (85\% vs. 51\% in PS) 
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Merging with Revelio Data}

\begin{frame}
    \frametitle{Merging Strategy}
    \textbf{Current Merge (66\% of Preferred Sample):} 
    \begin{itemize}
        \item Keep at company x lottery year level (inconsistent identifiers across years)
        \item Initial (tokenized) matching on company name
        \item Coarse filter on matches based on name similarity, name rarity, employer/worksite state, NAICS code (if reported)
        \item Matched firms submit more applications and lose slightly more on average than unmatched firms 
    \end{itemize}
    \textbf{Future Merge:}
    \begin{itemize}
        \item Include more detailed info on location 
        \item Probabilistic linking 
        \item Better de-duplication of both datasets
    \end{itemize}
\end{frame}

\section{Initial Results}
\begin{frame}
    \frametitle{Ideal Empirical Strategy and Outcomes of Interest}
    Following DGI, for firm $j$ in year $t$ who applied for an H-1B visa in fiscal year $T$, estimate: 
    \begin{equation}
        Y_{jtT} = \beta_0 + \beta_1U_{jT} + \varepsilon_{jtT}
    \end{equation}
    where $U_{jT}$ represents `excess' lottery wins above chance (i.e. number of lottery wins net of number of applications $\times$ aggregate win rate).

    \vspace{2mm}
    I am interested in the following outcomes:
    \begin{itemize}
        \item \textbf{First stage:} number of H-1B visas granted (requires additional application after winning)
        \item \textbf{Hiring:} number of workers hired; number of total workers
        \item \textbf{Workers:} average promotion rate of new hires; turnover of incumbent workers
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{First Pass at First Stage}
    Subset to firms that only submit one application in a given lottery year to simplify visualization:

        \item Balanced sample of firms from preferred sample that are matched and can be consistently and uniquely identified in the H-1B data in all four years 
        \item Subset to firms that submitted one application in 2022 (n = 1135)
        \item Plot mean number of H-1B visas approved
    \end{itemize}
\end{frame}
